    <script>
        // DOM elements
        const mainContent = document.getElementById('main-content');
        const statusBox = document.getElementById('status-box');
        const statusText = statusBox.querySelector('p');
        const videoElement = document.getElementById('video');
        const canvasElement = document.getElementById('output-canvas');
        const canvasCtx = canvasElement.getContext('2d');
        const eyeAlertBox = document.getElementById('eye-alert-box');

        // Posture Blur Functionality
        /**
         * Toggles the blur effect on the main content and updates the status message.
         * @param {boolean} shouldBlur - True to blur, false to unblur.
         */
        function toggleBlur(shouldBlur) {
            if (shouldBlur) {
                mainContent.classList.add('blur-active');
                statusText.textContent = 'Incorrect Posture! Please adjust.';
                statusBox.classList.remove('correct');
                statusBox.classList.add('incorrect');
            } else {
                mainContent.classList.remove('blur-active');
                statusText.textContent = 'Posture is Correct! Keep it up.';
                statusBox.classList.remove('incorrect');
                statusBox.classList.add('correct');
            }
        }

        // Eye Tracker (20-20-20 Rule) Functionality
        const WORK_INTERVAL = 20 * 60 * 1000;
        const REST_INTERVAL = 20 * 1000;
        let workTimer = null;
        let restTimer = null;
        let inRestPeriod = false;

        function showRestAlert() {
            eyeAlertBox.style.display = 'block';
        }

        function hideRestAlert() {
            eyeAlertBox.style.display = 'none';
        }

        function calculateEyeOpenness(landmarks, indices) {
            if (landmarks.length < Math.max(...indices) + 1) {
                return 0;
            }
            const upper = landmarks[indices[1]];
            const lower = landmarks[indices[5]];
            return Math.abs(upper.y - lower.y);
        }

        function startWorkTimer() {
            if (workTimer) clearTimeout(workTimer);
            workTimer = setTimeout(() => {
                inRestPeriod = true;
                showRestAlert();
                startRestTimer();
            }, WORK_INTERVAL);
        }

        function startRestTimer() {
            if (restTimer) clearTimeout(restTimer);
            restTimer = setTimeout(() => {
                inRestPeriod = false;
                hideRestAlert();
                startWorkTimer();
            }, REST_INTERVAL);
        }
        
        // Helper function to calculate the angle between three points
        function calculateAngle(p1, p2, p3) {
            const angle = Math.atan2(p3.y - p2.y, p3.x - p2.x) - Math.atan2(p1.y - p2.y, p1.x - p2.x);
            let degrees = Math.abs(angle * 180.0 / Math.PI);
            if (degrees > 180.0) {
                degrees = 360.0 - degrees;
            }
            return degrees;
        }

        // Main MediaPipe Logic and Integration
        /**
         * Processes the results from the MediaPipe Pose model.
         * @param {Object} results - The pose detection results.
         */
        function onPoseResults(results) {
            // Draw the canvas
            canvasCtx.save();
            canvasCtx.clearRect(0, 0, canvasElement.width, canvasElement.height);
            canvasCtx.drawImage(results.image, 0, 0, canvasElement.width, canvasElement.height);

            if (results.poseLandmarks) {
                const landmarks = results.poseLandmarks;
                
                // Ensure all required landmarks are present before proceeding
                if (!landmarks[11] || !landmarks[12] || !landmarks[23] || !landmarks[24]) {
                    toggleBlur(false);
                    canvasCtx.restore();
                    return;
                }
                
                // Get key landmarks for posture analysis
                const leftShoulder = landmarks[11];
                const rightShoulder = landmarks[12];
                const leftHip = landmarks[23];
                const rightHip = landmarks[24];
                
                // Calculate average shoulder and hip midpoints for a more robust check
                const shoulderMidpoint = { 
                    x: (leftShoulder.x + rightShoulder.x) / 2, 
                    y: (leftShoulder.y + rightShoulder.y) / 2 
                };
                const hipMidpoint = { 
                    x: (leftHip.x + rightHip.x) / 2, 
                    y: (leftHip.y + rightHip.y) / 2 
                };
                
                const torsoAngle = calculateAngle(hipMidpoint, shoulderMidpoint, {x: shoulderMidpoint.x, y: shoulderMidpoint.y - 1});
                
                const SLOUCH_THRESHOLD_DEGREES = 15;
                const isCorrectPosture = torsoAngle < SLOUCH_THRESHOLD_DEGREES;

                toggleBlur(!isCorrectPosture);
                
                // Define drawing styles based on posture
                const drawStyle = isCorrectPosture ? { color: '#10B981', lineWidth: 4 } : { color: '#EF4444', lineWidth: 4 };

                // Draw the pose on the canvas
                drawConnectors(canvasCtx, landmarks, POSE_CONNECTIONS, { color: drawStyle.color, lineWidth: drawStyle.lineWidth });
                drawLandmarks(canvasCtx, landmarks, { color: drawStyle.color, lineWidth: 2, radius: 4 });
            }
            canvasCtx.restore();
        }

        /**
         * Processes the results from the MediaPipe Face Mesh model.
         * @param {Object} results - The face mesh detection results.
         */
        function onFaceMeshResults(results) {
            if (inRestPeriod) return;

            if (results.multiFaceLandmarks && results.multiFaceLandmarks.length > 0) {
                const landmarks = results.multiFaceLandmarks[0];
                const leftEyeIndices = [159, 145, 33, 133, 173, 157];
                const rightEyeIndices = [386, 374, 263, 362, 398, 385];

                if (landmarks.length > Math.max(...leftEyeIndices, ...rightEyeIndices)) {
                    const leftEyeOpenness = calculateEyeOpenness(landmarks, leftEyeIndices);
                    const rightEyeOpenness = calculateEyeOpenness(landmarks, rightEyeIndices);
    
                    const threshold = 0.03;
    
                    if (leftEyeOpenness < threshold && rightEyeOpenness < threshold) {
                        startWorkTimer();
                    }
                }
            }
        }

        async function main() {
            // Initialize MediaPipe Pose
            const pose = new Pose({
                locateFile: (file) => `https://cdn.jsdelivr.net/npm/@mediapipe/pose@0.4/${file}`
            });
            pose.setOptions({
                modelComplexity: 1,
                smoothLandmarks: true,
                minDetectionConfidence: 0.5,
                minTrackingConfidence: 0.5,
            });
            pose.onResults(onPoseResults);

            // Initialize MediaPipe Face Mesh
            const faceMesh = new FaceMesh({
                locateFile: (file) => `https://cdn.jsdelivr.net/npm/@mediapipe/face_mesh@0.4/${file}`
            });
            faceMesh.setOptions({
                maxNumFaces: 1,
                refineLandmarks: true,
                minDetectionConfidence: 0.5,
                minTrackingConfidence: 0.5,
            });
            faceMesh.onResults(onFaceMeshResults);

            // Start the camera and send frames to both models
            const camera = new Camera(videoElement, {
                onFrame: async () => {
                    await pose.send({ image: videoElement });
                    await faceMesh.send({ image: videoElement });
                },
                width: 640,
                height: 480,
            });
            await camera.start();

            // Start the eye rest timer
            startWorkTimer();
        }

        main();
    </script>
